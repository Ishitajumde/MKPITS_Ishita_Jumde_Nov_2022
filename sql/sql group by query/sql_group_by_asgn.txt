1. Retrieve the total revenue generated for each film category.
Ans=select sum(payment.amount),film_category.category_id from payment 
join rental on rental.rental_id=payment.payment_id
join inventory on inventory.inventory_id=rental.inventory_id
join film on film.film_id=inventory.film_id
join film_category on film_category.category_id=film_category.film_id
group by film_category.category_id;

2. Determine the count of rentals for each customer.
Ans=   select count(rental_id) from rental 
join customer on rental.rental_id=customer.customer_id;

3. Calculate the average rental duration (in days) for each film.
 Ans= select avg(rental_duration)from film 
group by film_id;

4. Find the total revenue generated for each month.
Ans=select sum(amount),month(payment_date) from payment
group by month(payment_date);

5. Determine the total revenue generated by each store.
Ans=select sum(amount)from payment 
   join staff on staff.staff_id=payment.payment_id
   join  store on store.store_id=staff.store_id
   group by store.store_id;

6. Calculate the count of rentals handled by each staff member.
 Ans=  select count(rental_id),staff.staff_id,staff.first_name from rental join staff 
on staff.staff_id=rental.rental_id
group by rental_id;

7. Compute the average rental rate for each film category.
Ans=select avg(rental_rate) from film 
join film_category on film.film_id=film_category.category_id;

8. Determine the count of rentals for each film.
Ans=select count(rental_id),film_id,title from rental join film on                     rental.rental_id = film.film_id
     group by rental_id,film_id;

9. Retrieve the total revenue generated for each city.
 Ans=select sum(amount)from payment 
   join staff on staff.staff_id=payment.payment_id
   join  address on address.address_id=staff.address_id
   join city on city.city_id=address.city_id
   group by city.city_id;

10. Calculate the count of rentals for each film language.
  Ans= select count(rental_id)from rental 
join inventory on inventory.inventory_id=rental.rental_id
join film on film.film_id=inventory.film_id
join language on language.language_id=film.language_id
group by language.language_id;
